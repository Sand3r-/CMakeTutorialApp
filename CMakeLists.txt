cmake_minimum_required(VERSION 3.0.2)
project("neuron" CXX)
set(NEURON_TARGET neuron)
set(MATH_TARGET math_lib)
add_library(${MATH_TARGET} src/math.cpp)
add_executable (${NEURON_TARGET} src/main.cpp)
target_include_directories(${NEURON_TARGET} PRIVATE include)
target_include_directories(${MATH_TARGET} PRIVATE include)
add_dependencies(${NEURON_TARGET} ${MATH_TARGET})
target_link_libraries(${NEURON_TARGET} ${MATH_TARGET})
